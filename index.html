public_same_charge = []
private_same_charge = []
discount_schools = []

merged_df['discount_tuition'] = None

for index, row in merged_df.iterrows():
        
    # Get the name of the school out.
    name = row["name"]
    
    # Truth value.
    discount_tuition = True
    
    if (row["in_state_tuition"] >= row["out_of_state_tuition"]):    
        if (row["type"] == "Public"):
            public_same_charge.append(name)
        else:
            private_same_charge.append(name)
        
        discount_tuition = False
    else:
        discount_schools.append(name)
    
    # Set the truth value.
    merged_df.at[index, 'discount_tuition'] = discount_tuition    
    
print("The number of public schools that don't discount in-state tuition: " + str(len(public_same_charge)))
print("The number of private schools that don't discount in-state tuition: " + str(len(private_same_charge)))
print("The number of schools that discount in-state tuition: " + str(len(discount_schools)))
